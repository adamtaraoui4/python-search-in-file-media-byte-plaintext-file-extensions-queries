//We may use a multispectral sensor to get and set Celsius/fahrenheit rubber consortium data
;
long xUniverse = 1000; //example of the sum of the points of the universe is 1000.
long[] universePointsTemperatureDegree = new long[xUniverse]; //example of the universe points-atoms in an array
long coldestTemperatureToNeutralizeExtrraDifferentialOfRubber = -4; //variable of coldest point possible


//for loop to generate random(from 0 to 100) temperature points simulation, putting them in an array
//method generatingRandomDegreesInTheSimulatorOfTheUniverse(xUniverse, universePointsTemperatureDegree) to generate random degrees in the simulation of the universe.
static void generatingRandomDegreesInTheSimulatorOfTheUniverse(long xUniverse, long[] universePointsTemperatureDegree) { 
Random randomClassInstance = new Random();
long randomNumberGeneration;
for (long i =0;i<xUniverse;i++)
    {
    randomNumberGeneration = randomClassInstance.Next(100);
    universePointsTemperatureDegree[i] = randomNumberGeneration;
    }
}

//for loop method to print the elements of an array by their index
//method to print an array
static void printTemperatureOfThePointsOfTheUniverse(long xUniverse, long[] universePointsTemperatureDegree)
{
    for (long i = 0; i < xUniverse; i++)
    {
        Console.Write(universePointsTemperatureDegree[i] + " ");
    }
    Console.WriteLine("");
}


//for loop to neutralize the random temperature points simulated in the simulator, neutralizing the points upper than 4 degrees -4
//method generatingRandomDegreesInTheSimulatorOfTheUniverse(xUniverse, universePointsTemperatureDegree) to generate random degrees in the simulation of the universe.
static void neutralizingTheDifferentialRubber(long xUniverse, long[] universePointsTemperatureDegree, long coldestTemperatureToNeutralizeExtrraDifferentialOfRubber)
{
    for (long i = 0; i < xUniverse; i++)
    {
        if(universePointsTemperatureDegree[i] > 4)
        {
            universePointsTemperatureDegree[i] = -4;
        }
    }
}

generatingRandomDegreesInTheSimulatorOfTheUniverse(xUniverse,universePointsTemperatureDegree);
